def menu():
    # type: () -> object
    # type: () -> object
    print("Main Menu\n1: Fahrenheit to Celsius\n2: Celsius to Fahrenheit\n3: Exit Program\n")
    try:
        choice = int(input("Please enter 1, 2 or 3: "))
    except ValueError:
        print("Invalid entry")
        return runit()
    return choice


def celsiusToFahrenheit(temp):
    deg_fah = round(temp, 1)
    c_temp = round((temp - 32) * (5.0 / 9.0), 1)
    print deg_fah, "degrees Fahrenheit equals:", c_temp, "degrees Celsius"


def fahrenheitToCelsius(temp):
    deg_cel = round(temp, 1)
    f_temp = round((9.0 / 5.0) * (temp + 32), 1)
    print deg_cel, "degrees Celsius equals ", f_temp, "degrees Fahrenheit"


def getfahrenheit():
    global choice
    try:
        choice = float(input("Please enter degrees Fahrenheit:"))
    except ValueError:
        print("Invalid entry")
        runit()
    return fahrenheitToCelsius(choice)


def getcelsius():
    global choice
    try:
        choice = float(input("Please enter degrees Celsius:"))
    except ValueError:
        print("Invalid entry")
        runit()
    return celsiusToFahrenheit(choice)


def runit():
    loop = True
    while loop:
        path = menu()
        if path == 2:
            getfahrenheit()
        if path == 1:
            getcelsius()
        if path == 3:
            loop = False

print("Welcome to the CS 3030 Temperature Conversion Program\n\n\\")
